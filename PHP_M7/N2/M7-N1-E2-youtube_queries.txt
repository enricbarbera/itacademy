-- seleccionar videos que incorporin una determinada etiqueta ****************************
SELECT id_video, title, t.name AS 'tag name'
FROM videos v
JOIN videos_has_tags vht
ON vht.videos_id_video = v.id_video
JOIN tags t
ON vht.tags_id_tag = t.id_tag
-- WHERE vht.tags_id_tag = 1;
WHERE vht.tags_id_tag = 3;

-- seleccionar playlists contenint un determinat video (per id_video) ************************
SELECT  v.title AS 'video title', id_playlist, name AS 'playlist name'
FROM playlists p
JOIN included i
ON id_playlist = i.playlists_id_playlist
JOIN videos v
ON v.id_video = i.videos_id_video
-- WHERE i.videos_id_video = 3
WHERE i.videos_id_video = 4;

-- seleccionar canals seguits per determinats usuaris**************************************************
SELECT name AS 'channel name', description, u.username AS 'seguit per'
FROM channels c
JOIN follow f
ON id_channel = f.channels_id_channel
JOIN users u
ON id_user = f.users_id_user
-- WHERE follow.users_id_user IN (1,4)
WHERE f.users_id_user = 2
ORDER BY id_channel;
